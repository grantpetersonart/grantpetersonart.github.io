    function scrollFunction() {
    if (document.body.scrollTop > 100 || document.documentElement.scrollTop > 100) {
        document.getElementById("popnav").style.top = "0";
        document.getElementById("social").style.top = "0.25rem";
    } else {
        document.getElementById("popnav").style.top = "-100px";
        document.getElementById("social").style.top = "-100px";
    }
    }
